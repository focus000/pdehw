\begin{equation*}

\end{equation*}

\begin{equation*}
\tilde{u} = \frac{1}{1- \eta}\left( \int_{t_0}^1 - \int_{\eta}^{t_0} (\log{t} - \bar{u}_0) \right) \diff t
